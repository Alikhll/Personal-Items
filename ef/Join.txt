var a = (from user in db.Users
        join post in db.Posts
        on user.Id equals post.User_Id
        where post.User_Id == "c"
        select new
        {
        Id = user.Id,
        Cm = user.Name,
        Name = post.Context
        }).OrderBy(c=>c.Name).Skip(1).Take(1);
